document.addEventListener("DOMContentLoaded", () => {
  /**
   * Carrega arquivos HTML em elementos com o atributo [data-include],
   * garantindo que outros scripts possam aguardar sua conclus√£o.
   */
  async function loadIncludes() {
    const elements = document.querySelectorAll("[data-include]");

    // Cria um array de promises para cada opera√ß√£o de fetch.
    // O map sempre retornar√° uma promise, mesmo que a opera√ß√£o falhe,
    // permitindo que Promise.all aguarde a conclus√£o de todas.
    const promises = Array.from(elements).map(async (el) => {
      const file = el.dataset.include;
      if (!file) return;

      try {
        const response = await fetch(file);

        if (!response.ok) {
          // Lan√ßa um erro para ser capturado pelo bloco catch.
          throw new Error(`Falha na requisi√ß√£o com status ${response.status}`);
        }

        el.innerHTML = await response.text();
      } catch (error) {
        console.error(`Erro ao carregar o include: ${file}`, error);
        // Insere um coment√°rio no HTML para facilitar a depura√ß√£o.
        el.innerHTML = `<!-- Erro ao carregar ${file} -->`;
      }
    });

    // Aguarda que todos os includes (sucessos ou falhas) terminem.
    await Promise.all(promises);

    // Sinaliza globalmente que o processo foi conclu√≠do.
    window.includesReady = true;
    document.dispatchEvent(new CustomEvent("includes:ready"));
  }

  loadIncludes();
});
